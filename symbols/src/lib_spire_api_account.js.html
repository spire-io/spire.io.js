1 /**
  2  * @fileOverview Account Resource class definition
  3  */
  4 
  5 var Resource = require('./resource')
  6   ;
  7 
  8 /**
  9  * Represents an account in the spire api.
 10  *
 11  * @class Account resource
 12  *
 13  * @constructor
 14  * @extends Resource
 15  * @param spire {object} Spire object
 16  * @param data {object} Account data from the spire api
 17  */
 18 function Account(spire, data) {
 19   this.spire = spire;
 20   this.data = data;
 21   this.resourceName = 'account';
 22 }
 23 
 24 Account.prototype = new Resource();
 25 
 26 module.exports = Account;
 27 
 28 /**
 29  * Resets the account
 30  *
 31  * Note that this passes a session to the callback, and not an account.
 32  * This is because many of the session urls will have changed.
 33  *
 34  * @param {function (err, session)} cb Callback
 35  */
 36 Account.prototype.reset = function (cb) {
 37   var account = this;
 38   this.request('reset', function (err, sessionData) {
 39     if (err) return cb(err);
 40     account.data = sessionData.resources.account;
 41     cb(null, sessionData);
 42   });
 43 };
 44 
 45 /**
 46  * Updates the billing plan for the account.
 47  *
 48  * @param {object} info New billing plan data
 49  * @param {function (err, account)} cb Callback
 50  */
 51 Account.prototype.updateBillingSubscription = function (info, cb) {
 52   var account = this;
 53   this.request('update_billing_subscription', info, function (err, data) {
 54     if (err) return cb(err);
 55     account.data = data;
 56     cb(null, account);
 57   });
 58 };

 60 /**
 61  * Requests
 62  * These define API calls and have no side effects.  They can be run by calling
 63  *     this.request(<request name>);
 64  */
 65 
 66 /**
 67  * Reset your account
 68  * @name reset_key
 69  * @ignore
 70  */
 71 Resource.defineRequest(Account.prototype, 'reset', function () {
 72   return {
 73     method: 'post',
 74     url: this.url(),
 75     headers: {
 76       'Accept': this.mediaType(),
 77       'Authorization': this.authorization()
 78     }
 79   };
 80 });
 81 
 82 /**
 83  * Updates the billing subscription with the given data.
 84  * @name update_billing_subscription
 85  * @ignore
 86  */
 87 Resource.defineRequest(Account.prototype, 'update_billing_subscription', function (info) {
 88   var billing = this.data.billing;
 89   return {
 90     method: 'put',
 91     url: billing.url,
 92     content: info,
 93     headers: {
 94       'Accept': this.mediaType(),
 95       'Content-Type': this.mediaType(),
 96       'Authorization': this.authorization('invioces')
 97     }
 98   };
 99 });
100 
101 /**
102  * Gets the billing invoices for the account.
103  * @name billing_invoices
104  * @ignore
105  */
106 Resource.defineRequest(Account.prototype, 'billing_invoices', function () {
107   var invoices = this.data.billing.invoices;
108   return {
109     method: 'get',
110     url: invoices.url,
111     headers: {
112       'Accept': "application/json",
113       'Authorization': "Capability " + invoices.capability
114     }
115   };
116 });
117
