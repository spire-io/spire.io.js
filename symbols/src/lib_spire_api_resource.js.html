  8 /**
  9  * <p>Base class for objects repreenting resources in the spire api.  It is meant
 10  * to be extended by other classes.</p>
 11  *
 12  * <p>Resources have methods for making requests to the spire api.  These methods
 13  * are defined with `Request.defineRequest`.  Note that this is a method on the
 14  * Request object itself, not the prototype.  It is used to create methods on
 15  * the prototype of Resource classes.</p>
 16  *
 17  * <p>The `Resource` class provides default requests for the `get`, `put`,
 18  * `delete`, and `post` methods.  These methods can be overwritten by
 19  * subclasses.</p>
 20  *
 21  * <p>Once a request method has been defined, it can be run with
 22  * <pre><code>
 23  *   resource.request(<request name>);
 24  * </code></pre>
 25  * </p>
 26  *
 27  * <p>Such request methods have no side effects, and return JSON objects direct
 28  * from the spire api.</p>
 29  *
 30  * @constructor
 31  * @extends EventEmitter
 32  * @param {object} spire Spire object
 33  * @param {object} data Resource data from the spire api
 34  */
 35 function Resource (spire, data) {
 36   this.spire = spire;
 37   this.data = data;
 38 };
 39 
 40 Resource.prototype = new EventEmitter();
 41 
 42 module.exports = Resource;
 43 
 44 /**
 45  * Creates a request method on given object.
 46  *
 47  * `name` is the name of the request.  This is what gets passed to the `request`
 48  * method when actually calling the request.
 49  *
 50  * For instance, a request defined with
 51  * <pre><code>
 52  *   defineRequest(somePrototype, 'get', createGetReq);
 53  * </code></pre>
 54  *
 55  * is run by calling
 56  * <pre><code>
 57  *   resource.request('get', callback);
 58  * </code></pre>
 59  *
 60  * `fn` is a function that takes any number of arguments and returns a hash
 61  * describing the http request we are about to send.  Any arguments to this
 62  * function can be passed to the `request` method.
 63  *
 64  * For example, suppose `createGetReq` from above is the following function,
 65  * which takes an id number as argument as puts it in the query params:
 66  *
 67  * <pre><code>
 68  *   function createGetReq (id) {
 69  *     return {
 70  *       method: 'get',
 71  *       url: this.url(),
 72  *       query: { id: id },
 73  *      };
 74  *   };
 75  * </code></pre>
 76  *
 77  *  You would call that request like this:
 78  *  <code>resource.request('get', id, callback);</code>
 79  *
 80  * @param {object} obj Object to add the request method to.
 81  * @param {string} name Name of the request method
 82  * @param {function (*args)} fn Function taking any number of arguments, returning an object describing the HTTP request.
 83  */
 84 Resource.defineRequest = function (obj, name, fn) {
 85   obj['_req_' + name] = function () {
 86     var args = Array.prototype.slice.call(arguments);
 87     var callback = args.pop();
 88 
 89     var req = fn.apply(this, args)
 90 
 91     shred.request(req)
 92       .on('error', function (res) {
 93         var error = new ResponseError(res);
 94         callback(error);
 95       })
 96       .on('success', function (res) {
 97         callback(null, res.body.data);
 98       });
 99   };
100 };

102 /**
103  * <p>Makes the request with the given name.
104  *
105  * <p>Arguments are the request name, followed by any number of arguments that will
106  * be passed to the function which creates the request description, and a
107  * callback.
108  */
109 Resource.prototype.request = function () {
110   var args = Array.prototype.slice.call(arguments);
111   var reqName = args.shift();
112   var cb = args[args.length - 1];
113   if (typeof this['_req_' + reqName] !== 'function') {
114     return cb(new Error("No request defined for " + reqName));
115   }
116   this['_req_' + reqName].apply(this, args);
117 };

119 /**
120  * <p>Gets the resource.
121  *
122  * <p>Default method that may be overwritten by subclasses.
123  *
124  * @param {function (err, resource)} cd Callback
125  */
126 Resource.prototype.get = function (cb) {
127   var resource = this;
128   this.request('get', function (err, data) {
129     if (err) return cb(err);
130     resource.data = data;
131     cb(null, resource);
132   });
133 };

139  *
140  * @param {object} data Resource data
141  * @param {function (err, resource)} cd Callback
142  */
143 Resource.prototype.update = function (data, cb) {
144   var resource = this;

187 /**
188  * Returns the Authorization header for this resource, or for another capability
189  * if one is passed in.
190  *
191  * @param {Resource} [resource] Optional resource
192  * @returns {string} Authorization header
193  */
194 Resource.prototype.authorization = function (resource) {
195   var cap;
196   if (resource) {
197     if (typeof resource.capability === 'function') {
198       cap = resource.capability();
199     } else {
200       cap = resource.capability;
201     }
202   } else {
203     cap = this.capability();
204   }
205   return "Capability " + cap;
206 };
