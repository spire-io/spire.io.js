  5 var Resource = require('./resource')
  6   , _ = require('underscore')
  7   , async = require('async')

 10 /**
 11  * Represents a subscription in the spire api.
 12  *
 13  * <p>There are a few ways to get events from a subscription.
 14  *
 15  * <p>The first is to call <code>subscription.retrieveEvents</code> directly.
 16  * This is the most general method, and supports a number of options.
 17  *
 18  * <p>There are convenience methods <code>subscription.poll</code and
 19  * <code>subscription.longPoll</code> which wrap <code>retrieveEvents</code>.
 20  * The only difference is that <code>subscription.poll</code> has a timeout of
 21  * 0, so the request will always come back right away, while
 22  * <code>subscription.longPoll</code> has a 30 second timeout, so the request
 23  * will wait up to 30 seconds for new events to arrive before returning.
 24  *
 25  * <p>You can also use the <code>message</code> and <code>messages</code> events to
 26  * listen for new messages on the subscription.
 27  *
 28  * <p><pre><code>
 29  *    subscription.addListener('message', function (message) {
 30  *      console.log('Message received: ' + message.content);
 31  *    });
 32  *
 33  *    subscription.addListener('messages', function (messages) {
 34  *      console.log('Received ' + messages.length + ' messages.');
 35  *    });
 36  *
 37  *    subscription.startListening();
 38  * </code></pre>
 39  * </p>
 40  *
 41  * <p>The `messages` event fires first and contains the all the messages that were
 42  * received in a single request.  The `message` event fires once per message.
 43  *
 44  * @class Subscription Resource
 45  *
 46  * @constructor
 47  * @extends Resource
 48  * @param {object} spire Spire object
 49  * @param {object} data Subscription data from the spire api
 50  */
 51 function Subscription(spire, data) {
 52   this.spire = spire;
 53   this.data = data;
 54   this.resourceName = 'subscription';
 55 
 56   this.last = null;
 57   this.listening = false;
 58 }
 59 
 60 Subscription.prototype = new Resource();
 61 
 62 module.exports = Subscription;
 63 
 64 /**
 65  * Gets the name of the subscription.
 66  *
 67  * @returns {string} Name
 68  */
 69 Subscription.prototype.name = function () {
 70   return this.data.name;
 71 };

 73 /**
 74  * Starts long polling for the subscription.
 75  *
 76  * <p>The <code>message</code> and <code>messages</code> events will fire when a
 77  * request comes back with messages.  The <code>message</code> event will fire
 78  * once per message, while the <code>messages</code> event fires every time a
 79  * request comes back with more than one message.
 80  *
 81  * @example
 82  * subscription.addListener('message', function (message) {
 83  *   console.log('Message received: ' + message.content);
 84  * });
 85  *
 86  * subscription.addListener('messages', function (messages) {
 87  *   console.log('Received ' + messages.length + ' messages.');
 88  * });
 89  *
 90  * subscription.startListening();
 91  *
 92  * // Stop Listening after 100 seconds.
 93  * setTimout(function () {
 94  *   subscription.stopListening();
 95  *  }, 100000);
 96  *
 97  * @param {object} [options] Optional options argument
 98  * @param {number} [options.last] Optional last message
 99  * @param {number} [options.delay] Optional delay
100  * @param {number} [options.timeout] Optional timeout
101  * @param {string} [options.orderBy] Optional ordering ('asc' or 'desc')
102  * @param {function (err, messages)} cb Callback
103  */
104 Subscription.prototype.startListening = function (opts) {
105   this.listening = true;
106   this._listen(opts);
107 };
108 
109 /**
110  * Stops listening on the subscription.
111  */
112 Subscription.prototype.stopListening = function () {
113   this.listening = false;
114 };
115 
116 /**
117  * Gets events for the subscription.
118  *
119  * <p>This method only makes one request.  Use
120  * <code>subscription.startListening</code> to poll repeatedly.
121  *
122  * @example
123  * subscription.retrieveEvents(function (err, events) {
124  *   if (!err) {
125  *     // `events` is a hash with `messages`, `joins`, and `parts` (each possibly empty)
126  *   }
127  * });
128  *
129  * @param {object} [options] Optional options argument
130  * @param {number} [options.last] Optional last event
131  * @param {number} [options.delay] Optional delay
132  * @param {number} [options.timeout] Optional timeout
133  * @param {string} [options.orderBy] Optional ordering ('asc' or 'desc')
134  * @param {function (err, messages)} cb Callback
135  */
136 Subscription.prototype.retrieveEvents = function (options, cb) {
137   var subscription = this;
138   if (!cb) {
139     cb = options;
140     options = {};
141   }
142 
143   options.orderBy = options.orderBy || 'desc';
144 
145   var req = this.request('events', options, function (err, eventsData) {
146     if (err) {
147       subscription.emit('error', err);
148       return cb(err);
149     }
150 
151     if (!subscription.listening) {
152       return cb(null, eventsData);
153     }
154 
155     var eventTypes = ["messages", "joins", "parts"];
156 
157     _.each(eventTypes, function (eventType) {
158       if (eventsData[eventType].length) {
159         subscription.emit(eventType, eventsData[eventType])
160       }
161     });
162 
163     var allEvents = _.reduce(eventTypes, function (_allEvents, eventType) {
164       return _allEvents.concat(eventsData[eventType]);
165     }, []);
166 
167     allEvents.sort(function (a, b) {
168       return a.timestamp - b.timestamp
169     });
170 
171     if (allEvents.length) {
172       process.nextTick(function () {
173         _.each(allEvents, function (event) {
174           subscription.emit(event.type, event);
175         });
176       });
177     }
178 
179     cb(null, eventsData);
180   });
181 
182   req.on('socket', function (socket) {
183     subscription.emit('socket', socket);
184   });
185 };

192 /**
193  * Gets new events for the subscription.  This method forces a 0 second
194  * timeout, so the request will come back immediately, but may have an empty
195  * array of events if there are no new ones.
196  *
197  * <p>This method only makes one request.  Use
198  * <code>subscription.startListening</code> to poll repeatedly.
199  *
200  * @example
201  * subscription.poll(function (err, events) {
202  *   if (!err) {
203  *     // `events.messages` is an array of messages (possably empty)
204  *   }
205  * });
206  *
207  * @param {object} [options] Optional options argument
208  * @param {number} [options.delay] Optional delay
209  * @param {string} [options.orderBy] Optional ordering ('asc' or 'desc')
210  * @param {function (err, events)} cb Callback
211  */
212 Subscription.prototype.poll = function (options, cb) {
213   if (!cb) {
214     cb = options;
215     options = {};
216   }
217   options.timeout = 0;
218   this.longPoll(options, cb);
219 };
220 
221 /**
222  * Gets new events for the subscription.
223  *
224  * <p>This method defaults to a 30 second timeout, so the request will wait up to
225  * 30 seconds for a new message to come in.  You can increase the wait time with
226  * the <code>options.timeout</code> paraameter.
227  *
228  * <p>This method only makes one request.  Use `subscription.startListening` to
229  * poll repeatedly.
230  *
231  * @example
232  * subscription.longPoll({ timeout: 60 }, function (err, events) {
233  *   if (!err) {
234  *     // `events.messages` is an array of messages (possably empty)
235  *   }
236  * });
237  *
238  * @param {object} [options] Optional options argument
239  * @param {number} [options.delay] Optional delay
240  * @param {string} [options.orderBy] Optional ordering ('asc' or 'desc')
241  * @param {number} [options.timeout] Optional timeout
242  * @param {function (err, events)} cb Callback
243  */
244 Subscription.prototype.longPoll = function (options, cb) {
245   var subscription = this;
246   if (!cb) {
247     cb = options;
248     options = {};
249   }
250 
251   options.last = this.last;
252   options.timeout = options.timeout || 30;
253 
254   this.retrieveEvents(options, function (err, events) {
255     if (err) return cb(err);
256     if (events.last) {
257       subscription.last = events.last;
258     }
259 
260     cb(null, events);
261   });
262 };
263 
264 /**
265  * Repeatedly polls for new events until `subscription.stopListening` is
266  * called.
267  *
268  * You should use `subscription.startListening` instead of calling this method
269  * directly.
270  */
271 Subscription.prototype._listen = function (opts) {
272   var subscription = this;
273   opts = opts || {};
274   async.whilst(
275     function () { return subscription.listening; },
276     function (cb) {
277       subscription.longPoll(opts, cb);
278     },
279     function () {}
280   );
281 };

290 /**
291  * Gets the events for the subscription, according to various parameters.
292  * @name events
293  * @ignore
294  */
295 Resource.defineRequest(Subscription.prototype, 'events', function (options) {
296   options = options || {};
297   return {
298     method: 'get',
299     url: this.url(),
300     query: {
301       'timeout': options.timeout || 0,
302       'last': options.last || 0,
303       'order-by': options.orderBy || 'desc',
304       'delay': options.delay || 0
305     },
306     headers: {
307       'Authorization': this.authorization('events'),
308       'Accept': this.mediaType('events')
309     }
310   };
311 });
